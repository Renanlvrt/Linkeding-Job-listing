    <script>
        // --- UTILITY: Label Wrapping ---
        const wrapLabel = (label) => {
            if (label.length > 16) {
                return label.split(' '); // Simple word split for wrapping
            }
            return label;
        };

        // --- GLOBAL CHART DEFAULTS ---
        Chart.defaults.font.family = "'Inter', sans-serif";
        Chart.defaults.color = '#475569';
        
        // --- COMMON TOOLTIP CONFIG (MANDATORY) ---
        const commonTooltip = {
            callbacks: {
                title: function(tooltipItems) {
                    const item = tooltipItems[0];
                    let label = item.chart.data.labels[item.dataIndex];
                    if (Array.isArray(label)) {
                        return label.join(' ');
                    } else {
                        return label;
                    }
                }
            }
        };

        // --- CHART 1: THE FUNNEL (Bar) ---
        const ctxFunnel = document.getElementById('funnelChart').getContext('2d');
        new Chart(ctxFunnel, {
            type: 'bar',
            data: {
                labels: ['Total Scraped', 'Semantic Filter Pass', 'User Approved', 'Applications Sent'].map(wrapLabel),
                datasets: [{
                    label: 'Candidates',
                    data: [120, 45, 18, 15],
                    backgroundColor: [
                        '#94a3b8', // Slate 400
                        '#3b82f6', // Brand Blue
                        '#fbbf24', // Brand Gold
                        '#1e40af'  // Dark Blue
                    ],
                    borderRadius: 6
                }]
            },
            options: {
                responsive: true,
                maintainAspectRatio: false,
                plugins: {
                    legend: { display: false },
                    tooltip: commonTooltip
                },
                scales: {
                    y: { beginAtZero: true, grid: { display: false } },
                    x: { grid: { display: false } }
                }
            }
        });

        // --- CHART 2: SEMANTIC SCORING (Scatter) ---
        const ctxScatter = document.getElementById('scatterChart').getContext('2d');
        
        // Generate mock data for scatter plot
        const scatterData = [];
        for(let i=0; i<30; i++) {
            scatterData.push({x: i, y: Math.floor(Math.random() * 60) + 30}); // Low match noise
        }
        for(let i=30; i<40; i++) {
            scatterData.push({x: i, y: Math.floor(Math.random() * 25) + 75}); // High match hits
        }

        new Chart(ctxScatter, {
            type: 'scatter',
            data: {
                datasets: [{
                    label: 'Job Relevance Score',
                    data: scatterData,
                    backgroundColor: (ctx) => {
                        const v = ctx.raw?.y;
                        return v >= 70 ? '#fbbf24' : '#94a3b8'; // Gold for high match, Grey for low
                    },
                    pointRadius: 6,
                    pointHoverRadius: 8
                }]
            },
            options: {
                responsive: true,
                maintainAspectRatio: false,
                plugins: {
                    legend: { display: false },
                    tooltip: {
                        callbacks: {
                            label: function(context) {
                                return `Match Score: ${context.parsed.y}%`;
                            }
                        }
                    },
                    annotation: {
                        annotations: {
                            line1: {
                                type: 'line',
                                yMin: 70,
                                yMax: 70,
                                borderColor: '#1e40af',
                                borderWidth: 2,
                                borderDash: [5, 5],
                                label: {
                                    content: 'Threshold',
                                    enabled: true
                                }
                            }
                        }
                    }
                },
                scales: {
                    y: { 
                        min: 0, 
                        max: 100, 
                        title: { display: true, text: 'Semantic Match %' } 
                    },
                    x: { display: false } // Hide Job ID axis
                }
            }
        });

        // --- CHART 3: VELOCITY (Doughnut) ---
        const ctxVelocity = document.getElementById('velocityChart').getContext('2d');
        new Chart(ctxVelocity, {
            type: 'doughnut',
            data: {
                labels: ['Daily Limit Used', 'Remaining Capacity'],
                datasets: [{
                    data: [35, 15], // 35 used out of 50
                    backgroundColor: ['#1e40af', '#e2e8f0'],
                    borderWidth: 0,
                    cutout: '75%'
                }]
            },
            options: {
                responsive: true,
                maintainAspectRatio: false,
                plugins: {
                    legend: { display: false },
                    tooltip: commonTooltip
                }
            },
            plugins: [{
                id: 'textCenter',
                beforeDraw: function(chart) {
                    var width = chart.width,
                        height = chart.height,
                        ctx = chart.ctx;
            
                    ctx.restore();
                    var fontSize = (height / 114).toFixed(2);
                    ctx.font = "bold " + fontSize + "em sans-serif";
                    ctx.textBaseline = "middle";
                    ctx.fillStyle = "#1e40af";
            
                    var text = "35/50",
                        textX = Math.round((width - ctx.measureText(text).width) / 2),
                        textY = height / 2;
            
                    ctx.fillText(text, textX, textY);
                    ctx.save();
                }
            }]
        });
